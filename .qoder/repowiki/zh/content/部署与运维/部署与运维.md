## 简介
本文件为ZSTS-CMS项目的完整部署与运维指南，涵盖开发环境搭建、生产环境部署、系统迁移、故障排除、备份恢复和维护管理等全方位内容。项目采用前后端分离架构，包含AI内容生成前端（Next.js）、Node.js CMS后端和Go/Gin CMS后端三大部分，支持多页面内容管理、AI内容生成、视频/海报制作等功能。

## 故障排除指南

### 端口冲突解决

#### 开发环境端口清理
```bash
#!/bin/bash
# dev.sh中的端口清理功能
kill_port_if_listening() {
    local pids
    pids=$(ss -H -lntp 2>/dev/null | awk -v port="${DEPLOY_RUN_PORT}" '$4 ~ ":"port"$"' | grep -o 'pid=[0-9]*' | cut -d= -f2 | paste -sd' ' - || true)
    if [[ -z "${pids}" ]]; then
      echo "端口 ${DEPLOY_RUN_PORT} 空闲"
      return
    fi
    echo "端口 ${DEPLOY_RUN_PORT} 被PID: ${pids} 占用 (发送SIGKILL)"
    echo "${pids}" | xargs -I {} kill -9 {}
    sleep 1
    pids=$(ss -H -lntp 2>/dev/null | awk -v port="${DEPLOY_RUN_PORT}" '$4 ~ ":"port"$"' | grep -o 'pid=[0-9]*' | cut -d= -f2 | paste -sd' ' - || true)
    if [[ -n "${pids}" ]]; then
      echo "警告: 端口 ${DEPLOY_RUN_PORT} 仍被占用, PID: ${pids}"
    else
      echo "端口 ${DEPLOY_RUN_PORT} 已清理"
    fi
}
```

### 数据库备份

#### 自动化备份脚本
```bash
#!/bin/bash
# backup-database.sh
BACKUP_DIR="/var/backups/zsts-cms"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="zsts_cms"

mkdir -p $BACKUP_DIR

# 备份MySQL数据库
mysqldump -u root -p$MYSQL_PASSWORD $DB_NAME > $BACKUP_DIR/mysql_backup_$DATE.sql

# 压缩备份文件
gzip $BACKUP_DIR/mysql_backup_$DATE.sql

# 清理7天前的备份
find $BACKUP_DIR -name "*.sql.gz" -mtime +7 -delete

echo "数据库备份完成: $BACKUP_DIR/mysql_backup_$DATE.sql.gz"
```
